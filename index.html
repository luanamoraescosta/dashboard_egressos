Satisfação Geral com o Curso (escala de 0 a 10) 
{
  // Pega valores brutos, converte para número, arredonda para inteiro
  const raw = D.map(r => r.satisf_curso).filter(v => v != null && v !== "");
  const vals = raw.map(v => typeof v === "string" ? parseFloat(v) : +v).filter(x => !isNaN(x) && x >= 0 && x <= 10);
  
  // Conta cada nota inteira de 0 a 10
  const counts = {};
  for (let i = 0; i <= 10; i++) counts[i] = 0;
  vals.forEach(x => { counts[Math.round(x)] += 1; });
  
  const dd = [];
  for (let i = 0; i <= 10; i++) dd.push({nota: String(i), n: counts[i]});
  
  const avg = vals.length ? (vals.reduce((a,b)=>a+b) / vals.length) : 0;
  const sorted = vals.slice().sort((a,b)=>a-b);
  const med = sorted.length ? (sorted.length%2 ? sorted[Math.floor(sorted.length/2)] : (sorted[Math.floor(sorted.length/2)-1]+sorted[Math.floor(sorted.length/2)])/2) : 0;
  const maxN = Math.max(...dd.map(d=>d.n));

  const cn = {
    "0":"#991B1B","1":"#B91C1C","2":"#DC2626","3":"#EF4444","4":"#F87171",
    "5":"#FB923C","6":"#FBBF24","7":"#FCD34D","8":"#86EFAC","9":"#4ADE80","10":"#16A34A"
  };

  return htl.html`<div>
    <p style="margin:0 0 10px;font-size:.95rem;color:#374151">
      Média: <b style="color:#005BAC">${avg.toFixed(2)}</b> · 
      Mediana: <b style="color:#005BAC">${med.toFixed?med.toFixed(1):med}</b> · 
      Respondentes: <b>${F(vals.length)}</b>
    </p>
    ${Plot.plot({
      height: 380, width: 820,
      x: {label: "Nota atribuída", type: "band", domain: dd.map(d=>d.nota), padding: 0.2},
      y: {label: "Número de egressos", grid: true},
      marks: [
        Plot.barY(dd, {x: "nota", y: "n", fill: d => cn[d.nota] || "#999", tip: true}),
        Plot.text(dd.filter(d=>d.n>0), {x: "nota", y: "n", text: d => String(d.n), dy: -10, fontSize: 12, fontWeight: "600"}),
        Plot.ruleY([0])
      ]
    })}
  </div>`;
}

Avaliação Detalhada: Organização do Curso e Corpo Docente 
{
  const dims = [
    ["Organização geral do curso", "aval_organizacao_curso_num"],
    ["Disponibilidade da coordenação", "aval_coord_disponib_num"],
    ["Competência técnica dos professores", "aval_prof_competencia_num"],
    ["Didática dos professores", "aval_prof_didatica_num"],
    ["Disponibilidade dos professores", "aval_prof_disponib_num"]
  ];

  // Para cada dimensão, calcular distribuição completa + média
  const cats = [1, 2, 3, 4];
  const catLabels = {1:"Péssimo",2:"Regular",3:"Bom",4:"Ótimo"};
  const catCores = {1:VM, 2:AM, 3:"#60A5FA", 4:AZ};
  
  const rows = [];
  const medias = [];
  
  for (const [label, col] of dims) {
    const vals = V(D, col).map(Number).filter(x => !isNaN(x));
    const total = vals.length;
    const avg = total ? (vals.reduce((a,b)=>a+b)/total) : 0;
    medias.push({label, avg});
    
    const counts = {};
    vals.forEach(v => { const k = Math.round(v); counts[k] = (counts[k]||0)+1; });
    
    for (const c of cats) {
      const n = counts[c] || 0;
      const pct = total ? (n/total*100) : 0;
      rows.push({
        dim: label,
        cat: catLabels[c],
        catNum: c,
        n: n,
        pct: Math.round(pct*10)/10,
        color: catCores[c]
      });
    }
  }

  return htl.html`<div>
    <p style="margin:0 0 4px;font-size:.85rem;color:#6B7280">
      Escala: 1 = Péssimo, 2 = Regular, 3 = Bom, 4 = Ótimo
    </p>
    <div style="display:flex;gap:24px;flex-wrap:wrap;margin-bottom:12px">
      ${medias.map(m => htl.html`
        <div style="background:#f9fafb;border:1px solid #e5e7eb;border-radius:8px;padding:8px 14px;min-width:160px">
          <div style="font-size:.75rem;color:#6B7280">${m.label}</div>
          <div style="font-size:1.3rem;font-weight:700;color:${m.avg>=3?VD:m.avg>=2?AM:VM}">${m.avg.toFixed(2)}</div>
        </div>
      `)}
    </div>
    ${Plot.plot({
      marginLeft: 280, height: 350,
      x: {label: "% de egressos", grid: true},
      color: {
        domain: ["Péssimo","Regular","Bom","Ótimo"],
        range: [VM, AM, "#60A5FA", AZ],
        legend: true
      },
      marks: [
        Plot.barX(rows, {
          y: "dim", x: "pct", fill: "cat",
          offset: "normalize",
          tip: true,
          title: d => `${d.cat}: ${d.n} (${d.pct}%)`
        })
      ]
    })}
  </div>`;
}
Avaliação da Infraestrutura (filtrada pela seleção atual) 
{
  const dims = cfg.infra_dims;
  const cats = ["Péssimo","Regular","Bom","Ótimo"];
  const cors = {"Péssimo":VM,"Regular":AM,"Bom":"#60A5FA","Ótimo":AZ};
  
  const rows = [];
  for (const dim of dims) {
    const vals = V(D, dim.col);
    const total = vals.length;
    if (total === 0) continue;
    const counts = {};
    vals.forEach(v => { counts[v] = (counts[v]||0)+1; });
    for (const cat of cats) {
      const n = counts[cat] || 0;
      const pct = Math.round(n/total*1000)/10;
      rows.push({item: dim.label, cat, pct, n});
    }
  }
  
  return Plot.plot({
    marginLeft: 130, height: 500,
    x: {label: "% de respondentes", grid: true},
    color: {domain: cats, range: cats.map(c => cors[c]), legend: true},
    marks: [
      Plot.barX(rows, {
        y: "item", x: "pct", fill: "cat",
        offset: "normalize",
        tip: true,
        title: d => `${d.cat}: ${d.n} (${d.pct}%)`
      })
    ]
  });
}
cfg = FileAttachment("dados_dashboard.json").json()
DADOS = cfg.dados
P = cfg.paleta
AZ = P[0]; AM = P[1]; VD = P[2]; VM = P[3]
function V(d,c){return d.map(r=>r[c]).filter(v=>v!=null&&v!=="")}
function C(d,c){const o={};V(d,c).forEach(v=>{o[v]=(o[v]||0)+1});return o}
function A(d,c){const v=V(d,c).map(Number).filter(x=>!isNaN(x));return v.length?v.reduce((a,b)=>a+b)/v.length:0}
function M(d,c){const v=V(d,c).map(Number).filter(x=>!isNaN(x)).sort((a,b)=>a-b);if(!v.length)return 0;const m=Math.floor(v.length/2);return v.length%2?v[m]:(v[m-1]+v[m])/2}
function F(n){return Number(n).toLocaleString("pt-BR")}
function PC(a,b){return b?(a/b*100).toFixed(1)+"%":"--"}
viewof fc = Inputs.select(["", ...cfg.cursos_list], {label: "Curso"})
viewof fm = Inputs.select(["", ...cfg.municipios_list], {label: "Campus"})
viewof fs = Inputs.select(["", ...cfg.sexo_list], {label: "Gênero"})
viewof fr = Inputs.select(["", ...cfg.raca_list], {label: "Cor/Raça"})
viewof fb = Inputs.select(["", ...cfg.bolsa_list], {label: "Bolsa"})
viewof fe = Inputs.select(["", "Sim", "Não"], {label: "Empregado"})
viewof fa_min = Inputs.range([cfg.ano_min, cfg.ano_max], {label: "Ano de", step: 1, value: cfg.ano_min})
viewof fa_max = Inputs.range([cfg.ano_min, cfg.ano_max], {label: "Ano até", step: 1, value: cfg.ano_max})
D = DADOS.filter(r => {
  if (fc && r.curso !== fc) return false;
  if (fm && r.municipio_curso !== fm) return false;
  if (fs && r.sexo !== fs) return false;
  if (fr && r.cor_raca !== fr) return false;
  if (fb && r.bolsa_s !== fb) return false;
  if (fe && r.trabalhando_sn !== fe) return false;
  if (r.ano_conclusao != null) {
    const a = +r.ano_conclusao;
    if (a < fa_min || a > fa_max) return false;
  }
  return true;
})
